SELECT i.incident_title FROM incident i
INNER JOIN location l ON (i.location_id = l.id) 
INNER JOIN incident_category ic ON (ic.incident_id = i.id and ic.category_id = 10);
  
RIGHT JOIN incident_category c1 on c1.id = ic.category_id and c1.id = 10
RIGHT JOIN incident_category c2 on c2.id = ic.category_id and c2.id = 16;
WHERE (c1.id IS NOT NULL);
